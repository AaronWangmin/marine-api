244     * @param lon1 Origin longitude

246     * @param lon2 Destination longitude
247     * @return Distance in meters
248     */
249    private double haversine(double lat1, double lon1, double lat2, double lon2) {
250 
251        // Meridional Earth radius
252        // final double earthRadius = 6367.4491;
253        // a bit tweaked radius seems to produce more accurate results..?
254        final double earthRadius = 6366.70702;
255 
256        double dLat = Math.toRadians(lat2 - lat1);
257        double dLon = Math.toRadians(lon2 - lon1);
258 
259        double a = Math.sin(dLat / 2) * Math.sin(dLat / 2)
260                + Math.cos(Math.toRadians(lat1))
261                * Math.cos(Math.toRadians(lat2)) * Math.sin(dLon / 2)
262                * Math.sin(dLon / 2);
263 
264        double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
265 
266        double result = earthRadius * c * 1000;
267 
268        return result;
269    }
